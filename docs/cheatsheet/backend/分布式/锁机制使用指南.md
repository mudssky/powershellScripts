### 二、本地内存锁（同进程内多线程/协程竞争同一个变量）

#### 2.1 互斥锁（Mutex / Lock）

> **场景**：多个线程/协程同时读写同一个内存变量，需要排队执行。

**Python**

```python
# ========== 多线程场景（Django/Flask 等） ==========
import threading

lock = threading.Lock()
counter = 0

def increment():
    global counter
    with lock:            # 自动获取 & 释放
        counter += 1

# ========== 异步协程场景（FastAPI / asyncio） ==========
import asyncio

async_lock = asyncio.Lock()

async def async_increment():
    async with async_lock:  # 协程专用锁，不能混用 threading.Lock
        pass  # 操作共享状态
```

**Node.js (TypeScript)**

```typescript
// Node.js 是单线程，同步代码天然安全
// 只有 async/await 异步竞态才需要锁
import { Mutex } from 'async-mutex';

const mutex = new Mutex();
let counter = 0;

async function increment() {
    const release = await mutex.acquire();
    try {
        counter++;
    } finally {
        release();
    }
}

// 简写形式
async function incrementShort() {
    await mutex.runExclusive(async () => {
        counter++;
    });
}
```

**Go**

```go
import "sync"

var (
    mu      sync.Mutex
    counter int
)

func increment() {
    mu.Lock()         // 手动加锁
    defer mu.Unlock() // 推荐用 defer 确保释放
    counter++
}
```

**Java**

```java
// ========== 方式一：synchronized 关键字（最简单） ==========
private int counter = 0;

public synchronized void increment() {  // 整个方法加锁
    counter++;
}

public void incrementBlock() {
    synchronized (this) {  // 代码块级别加锁
        counter++;
    }
}

// ========== 方式二：ReentrantLock（更灵活，推荐） ==========
import java.util.concurrent.locks.ReentrantLock;

private final ReentrantLock lock = new ReentrantLock();
private int counter = 0;

public void increment() {
    lock.lock();
    try {
        counter++;
    } finally {
        lock.unlock();  // 必须在 finally 里释放
    }
}
```

#### 2.2 读写锁（RWLock）

> **场景**：读多写少的共享数据（如本地内存缓存）。多个读可以同时进行，写必须独占。

**Python**

```python
# 标准库无内置 RWLock，可用第三方库 readerwriterlock
from readerwriterlock import rwlock

rw = rwlock.RWLockFair()

def read_data():
    with rw.gen_rlock():   # 读锁，多个读操作可并发
        return shared_data

def write_data(value):
    with rw.gen_wlock():   # 写锁，独占
        shared_data = value
```

**Node.js (TypeScript)**

```typescript
// 通常用 async-rwlock 包
import ReadWriteLock from 'rwlock';

const rwlock = new ReadWriteLock();

function readData() {
    rwlock.readLock((release) => {
        console.log("读取中，其他读操作可并行...");
        release();
    });
}

function writeData() {
    rwlock.writeLock((release) => {
        console.log("写入中，独占...");
        release();
    });
}
```

**Go**

```go
import "sync"

var (
    rwMu  sync.RWMutex
    cache = make(map[string]string)
)

func read(key string) string {
    rwMu.RLock()         // 读锁
    defer rwMu.RUnlock()
    return cache[key]
}

func write(key, value string) {
    rwMu.Lock()          // 写锁（独占）
    defer rwMu.Unlock()
    cache[key] = value
}
```

**Java**

```java
import java.util.concurrent.locks.ReentrantReadWriteLock;

private final ReentrantReadWriteLock rwLock = new ReentrantReadWriteLock();
private final Map<String, String> cache = new HashMap<>();

public String read(String key) {
    rwLock.readLock().lock();
    try {
        return cache.get(key);
    } finally {
        rwLock.readLock().unlock();
    }
}

public void write(String key, String value) {
    rwLock.writeLock().lock();
    try {
        cache.put(key, value);
    } finally {
        rwLock.writeLock().unlock();
    }
}
```

#### 2.3 信号量（Semaphore）

> **场景**：限制同时并发执行某段逻辑的最大数量（如限制同时最多 5 个线程去调用第三方 API）。

**Python**

```python
import threading     # 多线程用 threading.Semaphore
import asyncio       # 协程用 asyncio.Semaphore

# ========== 多线程 ==========
sem = threading.Semaphore(5)  # 最多同时 5 个

def call_api():
    with sem:
        do_http_request()

# ========== 异步协程 ==========
async_sem = asyncio.Semaphore(5)

async def async_call_api():
    async with async_sem:
        await aiohttp_request()
```

**Node.js (TypeScript)**

```typescript
import { Semaphore } from 'async-mutex';

const semaphore = new Semaphore(5); // 最多 5 个并发

async function callApi() {
    const [value, release] = await semaphore.acquire();
    try {
        await fetch('https://api.example.com');
    } finally {
        release();
    }
}
```

**Go**

```go
// 使用带缓冲的 channel 实现信号量（Go 惯用法）
sem := make(chan struct{}, 5) // 最多同时 5 个

func callApi() {
    sem <- struct{}{}    // 获取信号量（满了就阻塞）
    defer func() { <-sem }()  // 释放信号量
    doHttpRequest()
}

// 或使用官方扩展包
// import "golang.org/x/sync/semaphore"
```

**Java**

```java
import java.util.concurrent.Semaphore;

private final Semaphore semaphore = new Semaphore(5);

public void callApi() throws InterruptedException {
    semaphore.acquire();  // 获取许可（超过 5 个就阻塞）
    try {
        doHttpRequest();
    } finally {
        semaphore.release();  // 释放许可
    }
}
```

### 三、数据库锁（保护数据库里的数据）

#### 3.1 乐观锁（Version 字段）

> **场景**：读多写少、冲突概率低的数据修改（扣库存、加积分等）。

**Python (SQLAlchemy)**

```python
class Product(Base):
    __tablename__ = 'product'
    id = Column(Integer, primary_key=True)
    stock = Column(Integer)
    version = Column(Integer, nullable=False)
    __mapper_args__ = {"version_id_col": version}  # 一行开启乐观锁

# 使用：正常 save 即可，冲突时自动抛出 StaleDataError
product.stock -= 1
session.commit()  # version 不一致 → StaleDataError
```

**Node.js (TypeORM)**

```typescript
@Entity()
class Product {
    @PrimaryGeneratedColumn()
    id: number;

    @Column()
    stock: number;

    @VersionColumn()             // 一行开启乐观锁
    version: number;
}

// 使用：正常 save 即可，冲突时抛出 OptimisticLockVersionMismatchError
product.stock -= 1;
await repo.save(product);
```

**Node.js (Prisma)**

```typescript
// Prisma 官方没有内置 @Version 装饰器，需要手动通过条件更新实现
// 1. 在 schema.prisma 中定义 version 字段
/*
model Product {
  id      Int @id @default(autoincrement())
  stock   Int
  version Int @default(1)
}
*/

// 2. 业务代码中使用 updateMany（update 仅支持唯一键作为条件）
const result = await prisma.product.updateMany({
    where: {
        id: productId,
        version: currentVersion, // 关键：更新前比对当前版本号
    },
    data: {
        stock: { decrement: 1 },
        version: { increment: 1 }, // 关键：成功后版本号 +1
    },
});

if (result.count === 0) {
    throw new Error("乐观锁冲突，数据已被修改");
}
```

**Go (GORM)**

```go
// GORM 无内置乐观锁，手动实现
result := db.Model(&Product{}).
    Where("id = ? AND version = ?", id, oldVersion).
    Updates(map[string]interface{}{
        "stock":   gorm.Expr("stock - 1"),
        "version": gorm.Expr("version + 1"),
    })

if result.RowsAffected == 0 {
    return errors.New("乐观锁冲突，请重试")
}
```

**Java (JPA/Hibernate)**

```java
@Entity
public class Product {
    @Id
    private Long id;
    private Integer stock;

    @Version                       // 一个注解开启乐观锁
    private Integer version;
}

// 使用：正常 save，冲突时抛出 OptimisticLockException
product.setStock(product.getStock() - 1);
entityManager.merge(product);
```

#### 3.2 悲观锁（FOR UPDATE）

> **场景**：金融级严谨操作、冲突概率极高的核心流程。

**Python (SQLAlchemy)**

```python
with session.begin():
    product = session.query(Product).with_for_update().get(1)
    product.stock -= 1
    session.commit()  # 提交事务时行锁释放
```

**Node.js (TypeORM)**

```typescript
await dataSource.transaction(async (manager) => {
    const product = await manager.findOne(Product, {
        where: { id: 1 },
        lock: { mode: "pessimistic_write" },
    });
    product.stock -= 1;
    await manager.save(product);
});
```

**Node.js (Prisma)**

```typescript
await prisma.$transaction(async (tx) => {
    // Prisma 需要借助原生 SQL（$queryRaw）实现行级悲观锁
    const result = await tx.$queryRaw`SELECT stock FROM "Product" WHERE id = ${1} FOR UPDATE`;

    if (result[0].stock <= 0) throw new Error("库存不足");

    await tx.product.update({
        where: { id: 1 },
        data: { stock: { decrement: 1 } },
    });
});
```

**Go (GORM)**

```go
tx := db.Begin()
var product Product
tx.Clauses(clause.Locking{Strength: "UPDATE"}).First(&product, 1)
product.Stock -= 1
tx.Save(&product)
tx.Commit()
```

**Java (JPA)**

```java
@Transactional
public void deductStock(Long id) {
    Product product = entityManager.find(
        Product.class, id, LockModeType.PESSIMISTIC_WRITE
    );
    product.setStock(product.getStock() - 1);
}
```

### 四、Redis 分布式锁（跨服务器/跨进程的资源竞争）

> **场景**：防止定时任务在多台服务器上同时执行、防接口重复提交、缓存击穿保护。

**Python (`redis-py`)**

```python
import redis
r = redis.Redis()

# 方式一：上下文管理器（推荐）
with r.lock("lock:order:1001", timeout=10, blocking_timeout=5):
    process_order()

# 方式二：手动控制
lock = r.lock("lock:order:1001", timeout=10)
if lock.acquire(blocking=False):       # 非阻塞模式，拿不到就立刻返回 False
    try:
        process_order()
    finally:
        lock.release()
```

**Node.js (`ioredis` + `redlock`)**

```typescript
import Redis from 'ioredis';
import Redlock from 'redlock';

const redis = new Redis();
const redlock = new Redlock([redis], { retryCount: 3 });

// 方式一：自动释放（推荐）
await redlock.using(['lock:order:1001'], 5000, async (signal) => {
    // signal.aborted 可以判断锁是否已经过期
    await processOrder();
});

// 方式二：手动控制
const lock = await redlock.acquire(['lock:order:1001'], 5000);
try {
    await processOrder();
} finally {
    await lock.release();
}
```

**Go (`go-redsync/redsync`)**

```go
import (
    "github.com/go-redsync/redsync/v4"
    "github.com/go-redsync/redsync/v4/redis/goredis/v9"
    goredis "github.com/redis/go-redis/v9"
)

client := goredis.NewClient(&goredis.Options{Addr: "localhost:6379"})
pool := goredis.NewPool(client)
rs := redsync.New(pool)

mutex := rs.NewMutex("lock:order:1001",
    redsync.WithExpiry(10*time.Second),
    redsync.WithTries(3),
)

if err := mutex.Lock(); err != nil {
    log.Fatal("获取锁失败", err)
}
defer mutex.Unlock()

processOrder()
```

**Java (`Redisson`)**

```java
import org.redisson.api.RLock;
import org.redisson.api.RedissonClient;

RLock lock = redisson.getLock("lock:order:1001");

// 方式一：自动看门狗续期（推荐）
lock.lock();   // 默认 30s，自动续期，不设超时
try {
    processOrder();
} finally {
    lock.unlock();
}

// 方式二：指定等待时间和过期时间
boolean acquired = lock.tryLock(
    5,    // 最多等待 5 秒
    10,   // 锁自动过期 10 秒
    TimeUnit.SECONDS
);
if (acquired) {
    try { processOrder(); }
    finally { lock.unlock(); }
}
```

### 五、文件锁（多进程写同一个本地文件）

> **场景**：多个进程写同一个日志文件、SQLite 数据库文件等。

| 语言 | 实现方式 | 代码 |
| :--- | :--- | :--- |
| **Python** | `filelock` 库 | `with FileLock("data.lock"): write_file()` |
| **Node.js** | `proper-lockfile` 库 | `await lockfile.lock('data.txt'); ...release()` |
| **Go** | `syscall.Flock` 或 `fslock` | `lock.Lock(); defer lock.Unlock()` |
| **Java** | `FileChannel.lock()` | `channel.lock(); try{...} finally{channel.close()}` |

> **核心记忆口诀：**
>
> - **内存里的数据** → 用**代码里的锁**（Mutex / RWLock）
> - **数据库里的数据** → 用**数据库自己的锁**（Version / FOR UPDATE）
> - **跨服务器的资源** → 用**Redis 分布式锁**
> - **Go 和 Java 写锁写到手软，Python 和 Node.js 基本把锁外包给了 Redis 和数据库**
